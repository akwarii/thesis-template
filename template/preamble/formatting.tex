% ----------------------------
%  Page Layout
% ----------------------------

% Custom formatting.
% Not applied to title page as it has its own geometry settings.
\geometry{%
    a4paper,
    top=2.5cm, bottom=2.5cm,
    left=3cm, right=2.5cm,
    bindingoffset=1cm,
    headheight=16pt,
    twoside,
}

% Line spacing
\renewcommand{\baselinestretch}{1.05}

% Paragraph formatting
\setlength{\parindent}{1.5em}     % Indentation
\setlength{\parskip}{0pt plus1pt} % Space between paragraphs

% ----------------------------
%  Headers & Footers
% ----------------------------

% Define fancy headers and footers
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}

\fancypagestyle{custom}{%
    \fancyhf{}
    \fancyhead[CE]{\scshape\footnotesize\leftmark}
    \fancyhead[CO]{\scshape\footnotesize\rightmark}
    \renewcommand{\headrulewidth}{0.4pt}
    \fancyfoot[C]{\thepage}
}

% Redefine plain page style to remove header and footer
\fancypagestyle{plain}{%
    \fancyhf{}
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0pt}
    \fancyfoot[C]{\thepage}
}

% Ensure new chapters start on an odd page
\let\oldchapter\chapter
\renewcommand{\chapter}{%
    \clearpage\thispagestyle{plain}
    \oldchapter
}

% ----------------------------
%  Table of Contents
% ----------------------------
\addto\captionsfrench{%
    \renewcommand{\contentsname}{Sommaire}
    \renewcommand{\listfigurename}{Liste des Figures}
    \renewcommand{\listtablename}{Liste des Tableaux}
}
\setcounter{tocdepth}{1}

% ----------------------------
%  Indexing & Glossary
% ----------------------------
\makeindex[columns=2, title=Index, intoc]

\newglossary[alg]{acronym}{acr}{acn}{\acronymname}
\newglossary[nlg]{notation}{not}{ntn}{Notations}
\makeglossaries
